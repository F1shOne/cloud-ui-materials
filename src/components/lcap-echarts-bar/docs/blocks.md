### 基本用法

``` html
<lcap-echarts-bar 
    xAxis="studentName" 
    yAxis="math, english, science" 
    xAxisTitle="姓名" 
    yAxisTitle="年龄/岁" 
    title="成绩统计"
    theme=""
    :titleFontSize=16
    titleFontStyle="italic"
    :allowDownload=false
    :allowShowLabel=false
    :allowShowHint=false
    :allowShowLegend=false
    :showXAxisLine=false
    :showYAxisLine=false
    :showXAxisLabel=true
    :showYAxisLabel=false
    :xAxisLabelRotate=0 ></lcap-echarts-bar>
```
