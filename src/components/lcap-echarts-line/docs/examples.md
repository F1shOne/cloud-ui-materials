### 基本用法

``` html
<lcap-echarts-line 
    xAxis="studentName" 
    yAxis="math, english, science" 
    xAxisTitle="姓名" 
    yAxisTitle="年龄/岁" 
    title="成绩统计"
    theme=""
    :titleFontSize=16
    titleFontStyle="italic"
    :allowDownload=true
    :allowShowLabel=true
    :allowShowHint=true
    :allowShowLegend=true
    :showXAxisLine=true
    :showYAxisLine=true
    :showXAxisLabel=true
    :showYAxisLabel=true
    :xAxisLabelRotate=0 >
</lcap-echarts-line>
```
