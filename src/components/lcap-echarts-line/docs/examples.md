### 基本用法

``` html
<lcap-echarts-line 
    xAxis="" 
    yAxis="" 
    xAxisTitle="维度" 
    yAxisTitle="指标" 
    title="标题"
    theme="theme1"
    :titleFontSize=16
    titleFontStyle="italic"
    :allowDownload=true
    :allowShowLabel=true
    :allowShowHint=true
    :allowShowLegend=true
    :showXAxisLine=true
    :showYAxisLine=true
    :showXAxisLabel=true
    :showYAxisLabel=true
    :xAxisLabelRotate=0 >
</lcap-echarts-line>
```
